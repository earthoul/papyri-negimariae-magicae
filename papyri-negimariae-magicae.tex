ad euocationem meam respondeat;
ego lege uniuersi inter me et amicum meum
uinculum adamantis immortalis magnifici animi iungam.
\magia{Vera Pactio.}%
\footnote{%
  Nonnulla verborum temporalium subj.\ coniugantur,
